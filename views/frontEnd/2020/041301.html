    <template>
  <div id="app">
    <canvas
      ref="canvas"
      class="canvas"
      width="400"
      height="400"
      @mousedown='mouseDown'
      @mouseup='mouseUp'
      @click="clickHandler">	
    </canvas>
  </div>
</template>

<script>
export default {
  data () {
    return {
      sDot: null,
      eDot: null
    }
  },
  methods: {
    mouseDown (e) {
      this.sDot = new Promise(resolve => {
        resolve([e.layerX, e.layerY])
      })
    },
    mouseUp (e) {
      this.eDot = new Promise(resolve => {
        resolve([e.layerX, e.layerY])
      })
    },
    drawCircle (sDot, eDot) {
      const canvas = this.$refs.canvas
      if (canvas.getContext) {
        const ctx = canvas.getContext("2d")
        const w = sDot[0] - eDot[0]
        const h = sDot[1] - eDot[1]
        const r = Math.sqrt(Math.pow(w, 2) + Math.pow(h, 2))
        ctx.beginPath()
        ctx.arc(sDot[0], sDot[1], r, 0, 2 * Math.PI)
        ctx.closePath()
        ctx.stroke()
      }
    },
    clickHandler () {
      Promise.all([this.sDot, this.eDot]).then(dotXY => {
        this.drawCircle(dotXY[0], dotXY[1])
      }).catch(err => {
        console.error(new Error(err))
      })
    }
  }
}
</script>
